There is nothing like subobject. An abstract class can have constructors also and the constructors are called when create an object of derived class. See following example: The program works fine and prints 10.


#include
using namespace std;

class Base {
int x;
public:
virtual void fun()=0;
Base(int i) { x = i; }
int getX() {return x;}
};

class Derived: public Base {
public:
Derived(int i):Base(i) {}
void fun() { }
};

int main()
{
Derived d(10);
cout << d.getX();
}
