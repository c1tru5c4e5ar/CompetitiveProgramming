node * reverseRecur(node *head)

{

if(head==null||head->next==null)

return head;

node *remain = head->next;

node *newnode = reverseRecur(remain);

head->next =null;

remain->next = head;

return newnode;

}

node * reverse(node **root, node *current)
{
node *tmp;
if (!current || !(*root))
return NULL;

tmp = current->next;

if (*root == current)
current->next = NULL;

tmp = reverse(root, tmp);

if (!tmp)
*root = current;
else
tmp->next = current;

return current;
}
